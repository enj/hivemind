# This is the mutect command for processing (CF33C,CF33P)
# in this example --intervals chr17:1-7577200 restricts the analysis to
# chromosome 17 from bp 1 to 7577200. Delete thius argument to analyze
# the complete exome

/usr/lib/jvm/jdk-6-oracle-x64/bin/java -Xmx8g -jar \
/opt/NGS/muTect/muTect-1.1.4.jar \
--analysis_type MuTect \
--reference_sequence /data/Annotation/GATK/hg19/ucsc.hg19.fasta \
--cosmic /data/Annotation/GATK/hg19/cosmic/Cosmic.hg19.vcf \
--dbsnp /data/Annotation/GATK/hg19/dbsnp_138.hg19.vcf \
--intervals chr17:1-7577200 \
--input_file:normal /ssddata/Patz/PROC/CF33/CF33C/CF33C-merged-recal.bam \
--input_file:tumor /ssddata/Patz/PROC/CF33/CF33P/CF33P-merged-recal.bam \
--out CF33-muTect-out.txt \
--coverage_file  CF33-muTect-coverage.txt



# See this 
http://gatkforums.broadinstitute.org/discussion/2226/cosmic-and-dbsnp-files-for-mutect

wget https://raw.githubusercontent.com/broadgsa/gatk/master/public/perl/sortByRef.pl
wget ftp://ngs.sanger.ac.uk/production/cosmic/CosmicCodingMuts_v68.vcf.gz
wget ftp://ngs.sanger.ac.uk/production/cosmic/CosmicNonCodingVariants_v68.vcf.gz
gunzip Cosmic*.gz
grep "^#" CosmicCodingMuts_v68.vcf > VCF_Header
grep -v "^#" CosmicCodingMuts_v68.vcf > Coding.clean
grep -v "^#" CosmicNonCodingVariants_v68.vcf > Noncoding.clean
cat Coding.clean Noncoding.clean | sort -gk 2,2 | awk '{print
"chr"$0}' | perl sortByRef.pl --k 1 -
/data/Annotation/GATK/hg19/ucsc.hg19.fasta.fai  > Cosmic.hg19
cat VCF_Header Cosmic.hg19 > Cosmic.hg19.vcf
